% ================================================================
%  plot_points.tex — Reusable macro to render a point cloud
%
%  \plotPoints[<options>]{<file.data>}
%
%  Options (all optional):
%    color   : fill & stroke color of the markers  (default: black)
%    size    : marker radius                        (default: 2pt)
%    opacity : fill opacity [0..1]                  (default: 0.85)
%
%  The data file must contain one "x  y" pair per line.
%  Lines starting with % are ignored (comments).
%
%  WHY \edef ON THE WHOLE CALL:
%    pgfplots defers ALL \addplot rendering until \end{axis}.
%    Using intermediate macros (\ppC, \ppS …) fails because they
%    are shared and overwritten by every subsequent \plotPoints call;
%    by render time every plot sees the last call's values.
%    \edef bakes the current color/size/opacity as literal strings
%    directly into the \addplot token list, so pgfplots receives
%    "colorA", "2pt", "0.85" — never a macro reference.
%
%  Example:
%    \plotPoints[color=blue!70!cyan, size=3pt]{cluster_a.data}
% ================================================================

%   color/.initial   = black,

%   opacity/.initial = 0.85,
